# In diese Datei befinden sich verschiedene Befehle zum aufzeichenen von verschiedenen Komponnenten.
# CPU Auslastung in einer Datei Aufzeichnen:

#!/bin/sh
i=1
while [ $i -le 5 ]
do
   stress -c 4 --timeout 10s
   cat /proc/stat | head -n1 >> cputest
  i=`expr $i + 1`
done

# RAM Auslastung in einer Datei Aufzeichnen:
i=1
while [ $i -le 5 ]
do
  stress -m 4 --timeout 10s
  vmstat | tail +3 >> ramtest
  i=`expr $i + 1`
done

# Scheduler und CPU Auslastung in einer Datei Aufzeichnen:
for ((i=0;i<=300;i++)); 
do 
mpstat | tail +4 >> schandcpurtest; 
done
